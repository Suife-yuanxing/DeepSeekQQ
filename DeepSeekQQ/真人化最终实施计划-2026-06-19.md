async def should_express_emotion(emotion, user_id) -> tuple[bool, str]:
    if emotion.intensity > 0.8:        # 高强度 → 很难隐藏
        return (True, "explicit")
    if emotion.intensity > 0.5:        # 中强度 → 40%隐藏
        if random.random() < 0.4: return (False, "hidden")
        if random.random() < 0.5: return (True, "micro")
        return (True, "explicit")
    if emotion.intensity > 0.2:        # 低强度 → 80%隐藏
